UVA11600 Masud Rana 题解

发表于 2022-02-27 更新于 2022-03-05 分类于 题解 阅读次数： 本文字数： 1.8k 阅读时长 ≈ 4 分钟

分析

一个好的思路就是只把有怪物的边看作边，能够不经过有怪物的边到达的点可以缩成一个点。

因为不经过有怪物的边到达的点，他们对答案产生的贡献是相同的。

问题就转化为了，期望走多少次，才能遍历缩点后的图。

设 \(f(S)\) 为当前经过的点集为 \(S\) 时，还需要的最小期望次数。

设当前集合 \(S\) 中共有 \(sz_S\) 个原图的点，点 \(j\) 中有 \(sz_j\) 个原图的点。 \[ f(S) = \frac{sz_S-1}{n-1} \cdot \big( f(S)+1 \big) + \sum_{j \notin S} {\frac{sz_j}{n-1} \cdot \big( f(S \cup \{j\}) + 1} \big) \] 显然求不出来这个式子，化简。

令 \(p = \frac{sz_S-1}{n-1}\)，\(p_j = \frac{sz_j}{n-1}\)。

则 \[ f(S) = p \cdot \big( f(S)+1 \big) + \sum_{j \notin S} {p_j \cdot \big( f(S \cup \{j\}) + 1} \big) \] \[ f(S) \cdot (1-p) = p + \sum_{j \notin S} {p_j \cdot \big( f(S \cup \{j\}) + 1} \big) \]

\[ f(S) = \frac{ p + (n-sz_S) + \sum_{j \notin S} f(S \cup \{ j \}) } {1-p} \]

代入，得 \[ f(S) = \frac{n-1}{n-sz_S} + \sum_{j \notin S} { f(S \cup \{j\}) \cdot \frac{sz_j}{n-sz_s}} \] 这个集合是不断扩大的，可以记忆化搜索。

复杂度 \(O(2^n)\)，状态数量有 \(2^n\) 个，数组开不下，可以用 map 存。

CODE

#include<cstdio>
#include<cstring>
#include<iostream>
#include<vector>
#include<map>
using namespace std;
#define pb push_back
const int N=35;
int t, _, n, m, num, sz[N];
// num 缩点后点的个数，sz[i] 缩点后第i个点包含的节点个数
bool v[N];
vector<int> w[N];
// 存没有怪物的边
map<int,double> f;
int dfs(int x) {
	v[x]=1;
	int sz=1;
	for(int i=0;i<w[x].size();++i) {
		int y=w[x][i];
		if(!v[y]) sz+=dfs(y);
	}
	return sz;
}
double dp(int s) {
	if(f[s]>1e-9) return f[s];
	int tot=0;
	for(int i=0;i<num;++i) if(s&(1<<i)) tot+=sz[i];
    // tot 集合s的节点个数
	if(tot==n) return f[s]=0;
	f[s]=(n-1)*1.0/(n-tot);
	for(int i=0;i<num;++i) if(!(s&(1<<i))) {
		f[s]+=dp(s|(1<<i))*sz[i]*1.0/(n-tot);
	}
	return f[s];
}
void sol() {
	num=0;
	memset(v,0,sizeof(v));
	memset(sz,0,sizeof(sz));
	scanf("%d%d",&n,&m);
	f.clear();
	for(int i=0;i<=n;++i) w[i].clear();
	for(int i=1;i<=m;++i) {
		int x, y;
		scanf("%d%d",&x,&y);
		w[x].pb(y), w[y].pb(x);
	}
	for(int i=1;i<=n;++i) if(!v[i]) sz[num++]=dfs(i);
    // 预处理，缩点
	printf("Case %d: %lf\n",++_,dp(1));
}
int main() { scanf("%d",&t); while(t--) sol(); }
