luogu2480 古代猪文 题解

发表于 2021-08-25 更新于 2022-03-26 分类于 题解 阅读次数： 本文字数： 1.5k 阅读时长 ≈ 3 分钟

link

求 \[ g^{\sum_{k \mid n}C_n^k} \bmod 9999911659 \] 显然的，那一长串的东西是质数。

设它为 $ p$ 好了。

则有 $ =1$

显然的，根据欧拉定理的推论可得 \[ g^{\sum_{k \mid n}{C_n^k}} \equiv g^{\sum_{k \mid n}{C_n^k \, \bmod \, p-1}} \quad (\bmod \, p) \] 所以本题就变成了求右边的那个东西。

$ p-1$ 显然不是个质数。

不难发现但很难算出 \[ p-1=2 \times 3 \times 4679 \times 35617 \] 可以用 $ O()$ 的时间枚举 $ n$ 的约数。

考虑到 \(n,g \le 10^9\)，我们用 $ lucas$ 定理求出 $ _{k n}{C_n^k}$，并记录它们对上述四个因子取模的结果，这样就得到了一个线性同余方程组。 \[ \begin{cases} x \bmod 2 = a_1 \\ x \bmod 3 = a_2 \\ x \bmod 4679 = a_3 \\ x \bmod 35617=a_4 \end{cases} \] 然后用中国剩余定理求出满足上述条件的最小的 \(\large x\)，用快速幂求 \(\large g^x\) 即可。

注意代码细节。

#include<cstdio>
#include<iostream>
using namespace std;
#define R register
#define ll long long
const ll a[4]={2,3,4679,35617};
ll n, g, ans, p=999911658ll, b[4], fac[36000]={1};
ll fp(ll x,ll y) {
    R ll z=1;
    for(;y;x=x*x%p,y>>=1ll) if(y&1ll) z=z*x%p;
    return z;
}
void exgcd(ll a,ll b,ll& x,ll& y) {
    if(!b) { x=1, y=0; return ; }
    exgcd(b,a%b,y,x), y-=a/b*x;
}
ll inv(ll a,ll p) {
    ll x, y;
    exgcd(a,p,x,y);
    return (x%p+p)%p;
}
ll C(ll x,ll p) {
    ll z=1, y=n, a, b;
    for(;x;x/=p,y/=p) {
        a=x%p, b=y%p;
        z=z*fac[b]%p*inv(fac[a],p)%p*inv(b<a?0:fac[b-a],p)%p;
    }
    return z;
}
int main() {
    int i, j;
    ll x, y;
    scanf("%lld%lld",&n,&g), g%=p+1;
    if(!g) { puts("0"); return 0; }
    for(i=1;i<=a[3];++i) fac[i]=fac[i-1]*i%p;
    for(i=1;i*i<=n;++i) if(n%i==0) {
        for(j=0;j<4;++j) b[j]=(b[j]+C(i,a[j]))%a[j];
        if(i*i!=n) for(j=0;j<4;++j) b[j]=(b[j]+C(n/i,a[j]))%a[j];
    }
    for(i=0;i<4;++i) {
        exgcd(p/a[i],a[i],x,y);
        ans=(ans+x*(p/a[i])%p*b[i])%p;
    }
    ans=(ans+p)%p, ++p;
    printf("%lld\n",fp(g,ans));
}
